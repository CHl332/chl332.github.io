Redis常用配置

daemonize no

pidfile “/var/run/redis/redis-server.pid”

port 6379

tcp-backlog 511

bind 192.168.98.70

unixsocket /var/run/redis/redis.sock

unixsocketperm 700

timeout 0

tcp-keepalive 0

loglevel notice

logfile “/var/log/redis/redis-server.log”

syslog-enabled no

syslog-ident redis

syslog-facility local0

databases 16

save 900 1

save 300 10

save 60 10000

save “”

stop-writes-on-bgsave-error yes

rdbcompression yes

rdbchecksum yes

dbfilename dump.rdb

dir ./
